    local function galatic127(bossName, difficulty)
        if not bossName or not galatic70(bossName) then return end
        if type(galatic66[bossName]) == "number" and galatic66[bossName] > (galatic63() and 2 or 1) then return end
        pcall(function()
            if bossName == "AtomicBoss" then
                galatic34.RemoteEvents.RequestSpawnAtomic:FireServer(difficulty)
            elseif bossName == "AnosBoss" then
                galatic34.Remotes.RequestSpawnAnosBoss:FireServer(galatic73(bossName), difficulty)
            elseif bossName == "RimuruBoss" then
                galatic34.RemoteEvents.RequestSpawnRimuru:FireServer(difficulty)
            elseif bossName == "TrueAizenBoss" then
                galatic34.RemoteEvents.RequestSpawnTrueAizen:FireServer(difficulty)
            elseif bossName == "GreatMageBoss" then
                galatic34.RemoteEvents.RequestSpawnGreatMage:FireServer(difficulty)
            elseif bossName == "TheWorldBoss" then
                galatic34.RemoteEvents.RequestSpawnTheWorld:FireServer(difficulty)
            elseif bossName == "SpiritWarriorBoss" then
                galatic34.RemoteEvents.RequestSpawnSpiritWarrior:FireServer(difficulty)
            elseif galatic71(bossName) then
                if bossName == "StrongestToday" or bossName == "StrongestHistory" then
                    galatic34.Remotes.RequestSpawnStrongestBoss:FireServer(bossName, difficulty)
                end
                galatic34.Remotes.RequestSummonBoss:FireServer(bossName, difficulty)
            else
                galatic34.Remotes.RequestSummonBoss:FireServer(bossName)
            end
        end)
    end

    local ConfigManager = {
        Folder = "SailorPiece",
        Options = {
            SaveManager_ConfigName = "",
            SaveManager_ConfigList = {}
        }
    }

    local function refreshConfigList()
        local configs = {}
        if isfolder and isfolder(ConfigManager.Folder .. "/settings") then
            for _, file in ipairs(listfiles(ConfigManager.Folder .. "/settings")) do
                if file:sub(-5) == ".json" then
                    local name = file:match("([^/]+)%.json$")
                    if name then
                        table.insert(configs, name)
                    end
                end
            end
        end
        table.sort(configs)
        return configs
    end

    local function loadConfig(name)
        if not name or name == "" then return false, "No config name provided" end
        local path = ConfigManager.Folder .. "/settings/" .. name .. ".json"
        if not isfile(path) then return false, "Config does not exist" end
        local success, data = pcall(readfile, path)
        if not success then return false, tostring(data) end
        local decoded = game:GetService("HttpService"):JSONDecode(data)
        for k, v in pairs(decoded) do
            if Settings[k] ~= nil then
                Settings[k] = v
            end
        end
        return true, "Config loaded"
    end

    local function saveConfig(name)
        if not name or name == "" then return false, "No config name provided" end
        local path = ConfigManager.Folder .. "/settings/" .. name .. ".json"
        local data = game:GetService("HttpService"):JSONEncode(Settings)
        if not writefile then return false, "File writing not supported" end
        local success, err = pcall(writefile, path, data)
        if not success then return false, tostring(err) end
        return true, "Config saved"
    end

    local function deleteConfig(name)
        if not name or name == "" then return false, "No config name provided" end
        local path = ConfigManager.Folder .. "/settings/" .. name .. ".json"
        if not isfile(path) then return false, "Config does not exist" end
        if not delfile then return false, "File deletion not supported" end
        local success, err = pcall(delfile, path)
        if not success then return false, tostring(err) end
        return true, "Config deleted"
    end

    local function setAutoload(name)
        if not name or name == "" then
            if isfile(ConfigManager.Folder .. "/settings/autoload.txt") then
                delfile(ConfigManager.Folder .. "/settings/autoload.txt")
            end
            return true, "Cleared autoload config"
        end
        local path = ConfigManager.Folder .. "/settings/autoload.txt"
        if not writefile then return false, "File writing not supported" end
        local success, err = pcall(writefile, path, name)
        if not success then return false, tostring(err) end
        return true, "Set autoload config"
    end

    local function getAutoload()
        if not isfile(ConfigManager.Folder .. "/settings/autoload.txt") then return nil end
        local success, data = pcall(readfile, ConfigManager.Folder .. "/settings/autoload.txt")
        if not success then return nil end
        return data
    end

                Fluent:Notify({Title = "Delete Config", Content = msg, Duration = 5})
                ConfigListDropdown:SetValues(refreshConfigList())
                ConfigListDropdown:SetValue("")
            else
                Fluent:Notify({Title = "Delete Config", Content = msg, Duration = 5})
            end
        end

                Fluent:Notify({Title = "Set Autoload", Content = "Select a config first", Duration = 5})

            Fluent:Notify({Title = "Set Autoload", Content = msg, Duration = 5})
        end

        Title = "Clear Autoload",
        Callback = function()
            local success, msg = setAutoload("")
            Fluent:Notify({Title = "Clear Autoload", Content = msg, Duration = 5})
        end

        loadConfig(autoloadConfig)

                Fluent:Notify({Title = "Key Verified", Content = "Key validated successfully!", Duration = 4})

        end

        Title = "Get Key",

            if setclipboard then

        Callback = function()

            Fluent:Notify({Title = "Link Copied", Content = "Paste the link on any browser to join the server!", Duration = 3})

    KeyTab:AddParagraph("KeyInfoParagraph1", {
